#include<stdio.h>

int f(unsigned int x);

int main() {
    unsigned int  i, j, a, b;
    while (scanf("%d%d", &i, &j) != EOF) {
        a = i;
        b = j;
        if (i > j) {
            int t = i;
            i = j;
            j = t;
        }
        int k, MaxCount = 0;
        for ( k = i; k <= j; k++) {
            int t = f(k);
            if (t > MaxCount)
                MaxCount = t;
        }
        printf("%d %d %d\n", a, b, MaxCount);
    }
    return 0;
}

int f(unsigned int x) {
    int count = 1;
    while (x != 1) {
        if (0 == x%2)
            x /= 2;
        else
            x = 3*x +1;
        count++;
    }
    return count;
}
