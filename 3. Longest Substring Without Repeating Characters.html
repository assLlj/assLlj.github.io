3. Longest Substring Without Repeating Characters

C

int lengthOfLongestSubstring(char * s){
    int dict[200] = {0};
    int result = 0;
    int i = 0, j = 0;
    while (s[i]) {
        memset(dict, 0, 200*sizeof(int));
        int count = 0;
        j = i;

        while (s[j]) {
            if (!dict[s[j]-' ']) {
                count++;
                dict[s[j]-' '] = 1;
            } else {
                break;
            }
            j++;
        }

        if (count > result)
            result = count;
        i++;
    }
    return result;
}
