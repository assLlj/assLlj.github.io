安装openvpn

opkg update
# 或者openvpn openvpn-polarssl
opkg install openvpn-ssl

配置

复制配置文件

mv my_openvpn.ovpn /etc/openvpn/.

创建新接口

vim /etc/config/network

添加：

config interface 'my_openvpn'
	option proto 'none'
		option ifname 'tun0'

创建新的Firewwall Zone

vim /etc/config/firewall

添加：

config zone
    option name 'openvpn_fw'
    option input 'REJECT'
    option output 'ACCEPT'
    option forward 'REJECT'
    option masq '1'
    option mtu_fix '1'
    option network 'my_openvpn'

config forwarding                               
        option dest 'openvpn_fw'                    
        option src 'lan'

重启服务

/etc/init.d/network restart
/etc/init.d/firewall restart

启动openvpn

openvpn --cd /etc/openvpn --config /etc/openvpn/my_openvpn.ovpn &

配置开机启动

vim /etc/rc.local

添加：

openvpn --cd /etc/openvpn --config /etc/openvpn/my_openvpn.ovpn &
